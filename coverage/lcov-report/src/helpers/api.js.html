export function liveDataRetrieval(id, accessToken) {
  return fetch(`${BASE_URL}api/ml/livedataretrieval`, {
    method: 'POST',
    body: JSON.stringify({
      id,
      u_token: accessToken,
    }),
    headers: defaultHeaders,
  }).then((response) => {
    if (response.status !== 200) {
      throw new Error(`Error Fetch Data Server Response Code${200}`);
    }
    return response.json();
  });
}

export function initBroadcast(broadcasterId, accessToken, title, tags) {
  return fetch(`${BASE_URL}api/ml/broadcast`, {
    method: 'POST',
    body: JSON.stringify({
      id: broadcasterId,
      u_token: accessToken,
      title,
      tag: tags,
    }),
    headers: defaultHeaders,
  }).then((response) => {
    if (response.status !== 200) {
      throw new Error(`Error creating broadcast with response code ${response.status}`);
    }
    return response.json();
  });
}
